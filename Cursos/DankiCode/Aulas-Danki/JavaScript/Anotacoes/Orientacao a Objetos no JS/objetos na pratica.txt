function Usuario(nome, idade, cargo) {
    this.nome = nome;
    this.idade = idade;
    this.cargo = cargo;
    this.apresentar = function () {
        console.log(`Saudações, meu nome é ${this.nome}, tenho ${this.idade} anos de idade e atualmente sou ${this.cargo} da empresa.`)
    }
}

let estagiario_1 = new Usuario('Felippe','25','Gerente Financeiro')

estagiario_1.apresentar()
