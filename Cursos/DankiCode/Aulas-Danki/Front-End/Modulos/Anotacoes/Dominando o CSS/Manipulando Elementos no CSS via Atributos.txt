*{
    margin: 0;
    padding: 0;
    box-sizing: border-box;
}
html,body{
    height: 100%;
}

a[href]{
    background-color: yellow;
}/*SINTAXE PARA ATACAR TODOS OS LINKS COM ATRIBUTO HREF*/
a[href="http://www.google.com"]{
    background-color: grey;
}/*SENDO MAIS ESPECÍFICO DE QUAL ATRIBUTO ATACAR SEM COLOCAR CLASS*/
a[href*="google"]{
    background-color: chartreuse;
}/* (A[HREF*="GOOGLE"]) VAI ATAVAR TODOS OS LINKS QUE TENHA A 
PALAVRA GOOGLE*/


input{
    background-color: yellow;
}/*ATAQUEI TODOS OS INPUTS*/
input[type=text]{
    background-color: green;
}/*FUI MAIS ESPECÍFICO NO ATRIBUTO QUE QUERIA ATACAR*/
