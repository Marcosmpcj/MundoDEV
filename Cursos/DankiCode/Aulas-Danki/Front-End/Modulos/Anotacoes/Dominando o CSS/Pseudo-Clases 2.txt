/*Exemplo 2*/
.box{
    width: 100%;
}
.item{
    float: left;
    width: 33.3%;
    height: 400px;
    background-color: red;
    border: 1px black solid;
}
.box div:last-of-type{
background-color: blue;

}

/*
>list-style-position: inside= exibir as marcações da lista;
>nth-of-type(): ataca a ordem de acordo com o tipo de elemento
 sem a necessidade de aplicar classes;
>nth-child: ataca a ordem da posição dos elementos filhos independente
do tipo ;
.box div:last-of-type= altera características do último elemento;
.box div:not(.one)= negação;
*/
