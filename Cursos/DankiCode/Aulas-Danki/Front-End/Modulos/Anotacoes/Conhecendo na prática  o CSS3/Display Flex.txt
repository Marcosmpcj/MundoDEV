 /*
 ==> PROPRIEDADES USADAS NO PARENT
 >flex-direction: row; = itens dispostos em linha ;
 >flex-direction: row-reverse; = itens dispostos em linha e invertidos;
 >flex-direction:column; = itens dispostos em colunas;
 >flex-direction:column-reverse; = itens dispostos em colunas e invertidos;
 >flex-wrap: wrap; = repeita a quebra de linha, logo quand necessário;
 >flex-wrap: wrap-reverse; = repeita a quebra de linha, logo quand necessário e inverte os itens;
 >flex-flow: column/row wrap ; = junção das duas funcionalidades do wrap e direction;
 >justify-content: center; = alinha os itens horizontalmente no eixo-X em relação ao parent;
 >justify-content: space-between; = alinha horizontalmente com o mesmo espaçamento entre os itens, sem os lados;
 >justify-content: space-around; = alinha horizontalmente com  espaçamento ao redor dos itens;
 >justify-content: space-evenly; = alinha horizontalmente com o mesmo espaçamento considerando todas as laterais;
 >align-item: center; = alinha os itens verticalmente no eixo-Y em relação ao parent;
 ## gap: (linha)px (coluna)px ; = atribui um espaçamento entre os itens, tipo uma margem;

  ==> PROPRIEDADES USADAS NO CHILD

>flex-grow: 1 ; = aumenta a qtd de espaço disponível para uso que o item ocupa;
>flex-shrink: 2 ; = diminui a qtd de espaço disponível para uso que o item ocupa;
>flex-basis: 100px ; = atribui um valor específico para o tamanho de um item;
>flex: ; = junta as propriedade do (grow/shrink/basis) nesta ordem; 
>align-self: ; = alinha um item específico na vertical;
>order: 3 ; = altera a ordem dos itens da posição inicial;
*/
